46  /**
47   * JSON "diff" implementation
48   *
49   * <p>This class generates a JSON Patch (as in, an RFC 6902 JSON Patch) given
50   * two JSON values as inputs. The patch can be obtained directly as a {@link
51   * JsonPatch} or as a {@link JsonNode}.</p>
52   *
53   * <p>Note: there is <b>no guarantee</b> about the usability of the generated
54   * patch for any other source/target combination than the one used to generate
55   * the patch.</p>
56   *
57   * <p>This class always performs operations in the following order: removals,
58   * additions and replacements. It then factors removal/addition pairs into
59   * move operations, or copy operations if a common element exists, at the same
60   * {@link JsonPointer pointer}, in both the source and destination.</p>
61   *
62   * <p>You can obtain a diff either as a {@link JsonPatch} directly or, for
63   * backwards compatibility, as a {@link JsonNode}.</p>
64   *
65   * @since 1.2
66   */
67  @ParametersAreNonnullByDefault
68  public final class JsonDiff
69  {
70      private static final ObjectMapper MAPPER = new ObjectMapper();
71  
72      private static final Equivalence<JsonNode> EQUIVALENCE
73          = JsonNumEquals.getInstance();
74  
75      private static final JsonPointer EMPTY_JSON_POINTER = JsonPointer.compile("");
76  
77      private JsonDiff()
78      {
79      }
80  
81      public static JsonPatch asJsonPatch(final JsonNode source, final JsonNode target) {
82          return asJsonPatch(source, target, ReplaceMode.RFC6902);
83      }
84  
85      /**
86       * Generate a JSON patch for transforming the source node into the target
87       * node
88       *
89       * @param source the node to be patched
90       * @param target the expected result after applying the patch
91       * @param replaceMode the replace mode to be used
92       * @return the patch as a {@link JsonPatch}
93       *
94       * @since 1.9
95       */
96      public static JsonPatch asJsonPatch(final JsonNode source,
97          final JsonNode target, ReplaceMode replaceMode)
98      {
99          requireNonNull(source, "source");
100         requireNonNull(target, "target");
101         final Map<JsonPointer, JsonNode> unchanged
102             = getUnchangedValues(source, target);
103         final DiffProcessor processor = new DiffProcessor(unchanged, replaceMode);
104 
105         generateDiffs(processor, EMPTY_JSON_POINTER, source, target);
106         return processor.getPatch();
107     }
108 
109     public static JsonNode asJson(final JsonNode source, final JsonNode target) {
110         return asJson(source, target, ReplaceMode.RFC6902);
111     }
112 
113     /**
114      * Generate a JSON patch for transforming the source node into the target
115      * node
116      *
117      * @param source the node to be patched
118      * @param target the expected result after applying the patch
119      * @param replaceMode the replace mode to be used
120      * @return the patch as a {@link JsonNode}
121      */
122     public static JsonNode asJson(final JsonNode source, final JsonNode target, ReplaceMode replaceMode)
123     {
124         final String s;
125         try {
126             s = MAPPER.writeValueAsString(asJsonPatch(source, target, replaceMode));
127             return MAPPER.readTree(s);
128         } catch (IOException e) {
129             throw new RuntimeException("cannot generate JSON diff", e);
130         }
131     }
132 
133     private static void generateDiffs(final DiffProcessor processor,
134         final JsonPointer pointer, final JsonNode source, final JsonNode target)
135     {
136         if (EQUIVALENCE.equivalent(source, target))
137             return;
138 
139         final JsonNodeType firstType = source.getNodeType();
140         final JsonNodeType secondType = target.getNodeType();
141 
142         /*
143          * Node types differ: generate a replacement operation.
144          */
145         if (firstType != secondType) {
146             processor.valueReplaced(pointer, source, target);
147             return;
148         }
149 
150         /*
151          * If we reach this point, it means that both nodes are the same type,
152          * but are not equivalent.
153          *
154          * If this is not a container, generate a replace operation.
155          */
156         if (!source.isContainerNode()) {
157             processor.valueReplaced(pointer, source, target);
158             return;
159         }
160 
161         /*
162          * If we reach this point, both nodes are either objects or arrays;
163          * delegate.
164          */
165         if (firstType == JsonNodeType.OBJECT)
166             generateObjectDiffs(processor, pointer, (ObjectNode) source,
167                 (ObjectNode) target);
168         else // array
169             generateArrayDiffs(processor, pointer, (ArrayNode) source,
170                 (ArrayNode) target);
171     }
172 
173     private static void generateObjectDiffs(final DiffProcessor processor,
174         final JsonPointer pointer, final ObjectNode source,
175         final ObjectNode target)
176     {
177         final Set<String> firstFields
178             = Sets.newTreeSet(Sets.newHashSet(source.fieldNames()));
179         final Set<String> secondFields
180             = Sets.newTreeSet(Sets.newHashSet(target.fieldNames()));
181 
182         for (final String field: Sets.difference(firstFields, secondFields))
183             processor.valueRemoved(pointer.append(JsonPointer.valueOf("/" + field)), source.get(field));
184 
185         for (final String field: Sets.difference(secondFields, firstFields))
186             processor.valueAdded(pointer.append(JsonPointer.valueOf("/" + field)), target.get(field));
187 
188         for (final String field: Sets.intersection(firstFields, secondFields))
189             generateDiffs(processor, pointer.append(JsonPointer.valueOf("/" + field)), source.get(field),
190                 target.get(field));
191     }
192 
193     private static void generateArrayDiffs(final DiffProcessor processor,
194         final JsonPointer pointer, final ArrayNode source,
195         final ArrayNode target)
196     {
197         final int firstSize = source.size();
198         final int secondSize = target.size();
199         final int size = Math.min(firstSize, secondSize);
200 
201         /*
202          * Source array is larger; in this case, elements are removed from the
203          * target; the index of removal is always the original arrays's length.
204          */
205         for (int index = size; index < firstSize; index++)
206             processor.valueRemoved(pointer.append(JsonPointer.valueOf("/" + size)), source.get(index));
207 
208         for (int index = 0; index < size; index++)
209             generateDiffs(processor, pointer.append(JsonPointer.valueOf("/" + index)), source.get(index),
210                 target.get(index));
211 
212 
213         // Deal with the destination array being larger...
214         JsonPointer endOfArrayPointer = JsonPointer.valueOf("/-");
215         for (int index = size; index < secondSize; index++)
216             processor.valueAdded(pointer.append(endOfArrayPointer), target.get(index));
217     }

220     @VisibleForTesting
221     static Map<JsonPointer, JsonNode> getUnchangedValues(final JsonNode source,
222         final JsonNode target)
223     {
224         final Map<JsonPointer, JsonNode> ret = Maps.newHashMap();
225         computeUnchanged(ret, EMPTY_JSON_POINTER, source, target);
226         return ret;
227     }
228 
229     private static void computeUnchanged(final Map<JsonPointer, JsonNode> ret,
230         final JsonPointer pointer, final JsonNode first, final JsonNode second)
231     {
232         if (EQUIVALENCE.equivalent(first, second)) {
233             ret.put(pointer, second);
234             return;
235         }
236 
237         final JsonNodeType firstType = first.getNodeType();
238         final JsonNodeType secondType = second.getNodeType();
239 
240         if (firstType != secondType)
241             return; // nothing in common
242 
243         // We know they are both the same type, so...
244 
245         switch (firstType) {
246             case OBJECT:
247                 computeObject(ret, pointer, first, second);
248                 break;
249             case ARRAY:
250                 computeArray(ret, pointer, first, second);
251             default:
252                 /* nothing */
253         }
254     }
255 
256     private static void computeObject(final Map<JsonPointer, JsonNode> ret,
257         final JsonPointer pointer, final JsonNode source,
258         final JsonNode target)
259     {
260         final Iterator<String> firstFields = source.fieldNames();
261 
262         String name;
263 
264         while (firstFields.hasNext()) {
265             name = firstFields.next();
266             if (!target.has(name))
267                 continue;
268             computeUnchanged(ret, pointer.append(JsonPointer.valueOf("/" + name)), source.get(name),
269                 target.get(name));
270         }
271     }
272 
273     private static void computeArray(final Map<JsonPointer, JsonNode> ret,
274         final JsonPointer pointer, final JsonNode source, final JsonNode target)
275     {
276         final int size = Math.min(source.size(), target.size());
277 
278         for (int i = 0; i < size; i++)
279             computeUnchanged(ret, pointer.append(JsonPointer.valueOf("/" + i)), source.get(i),
280                 target.get(i));
281     }
