20  package com.linecorp.centraldogma.internal.jsonpatch;
21  
22  import static com.fasterxml.jackson.annotation.JsonSubTypes.Type;
23  import static com.fasterxml.jackson.annotation.JsonTypeInfo.As;
24  import static com.fasterxml.jackson.annotation.JsonTypeInfo.Id;
25  
26  import com.fasterxml.jackson.annotation.JsonIgnoreProperties;
27  import com.fasterxml.jackson.annotation.JsonSubTypes;
28  import com.fasterxml.jackson.annotation.JsonTypeInfo;
29  import com.fasterxml.jackson.core.JsonPointer;
30  import com.fasterxml.jackson.databind.JsonNode;
31  import com.fasterxml.jackson.databind.JsonSerializable;
32  
33  @JsonTypeInfo(use = Id.NAME, include = As.PROPERTY, property = "op")
34  
35  @JsonSubTypes({
36      @Type(name = "add", value = AddOperation.class),
37      @Type(name = "copy", value = CopyOperation.class),
38      @Type(name = "move", value = MoveOperation.class),
39      @Type(name = "remove", value = RemoveOperation.class),
40      @Type(name = "replace", value = ReplaceOperation.class),
41      @Type(name = "safeReplace", value = SafeReplaceOperation.class),
42      @Type(name = "test", value = TestOperation.class)
43  })
44  
45  /**
46   * Base abstract class for one patch operation
47   *
48   * <p>Two more abstract classes extend this one according to the arguments of
49   * the operation:</p>
50   *
51   * <ul>
52   *     <li>{@link DualPathOperation} for operations taking a second pointer as
53   *     an argument ({@code copy} and {@code move});</li>
54   *     <li>{@link PathValueOperation} for operations taking a value as an
55   *     argument ({@code add}, {@code replace} and {@code test}).</li>
56   * </ul>
57   */
58  @JsonIgnoreProperties(ignoreUnknown = true)
59  public abstract class JsonPatchOperation
60      implements JsonSerializable
61  {
62      protected final String op;
63  
64      /*
65       * Note: no need for a custom deserializer, Jackson will try and find a
66       * constructor with a single string argument and use it.
67       *
68       * However, we need to serialize using .toString().
69       */
70      protected final JsonPointer path;
71  
72      /**
73       * Constructor
74       *
75       * @param op the operation name
76       * @param path the JSON Pointer for this operation
77       */
78      protected JsonPatchOperation(final String op, final JsonPointer path)
79      {
80          this.op = op;
81          this.path = path;
82      }
83  
84      /**
85       * Apply this operation to a JSON value
86       *
87       * @param node the value to patch
88       * @return the patched value
89       * @throws JsonPatchException operation failed to apply to this value
90       */
91      public abstract JsonNode apply(final JsonNode node)
92          throws JsonPatchException;
93  
94      @Override
95      public abstract String toString();
96  }
