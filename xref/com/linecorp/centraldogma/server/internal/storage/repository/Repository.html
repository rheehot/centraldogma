72      /**
73       * Returns {@code true} if and only if an {@link Entry} exists at the specified {@code path}.
74       */
75      default CompletableFuture<Boolean> exists(Revision revision, String path) {
76          validateFilePath(path, "path");
77          return find(revision, path, EXISTS_FIND_OPTIONS).thenApply(result -> !result.isEmpty());
78      }
79  
80      /**
81       * Retrieves an {@link Entry} at the specified {@code path}.
82       *
83       * @throws EntryNotFoundException if there's no entry at the specified {@code path}
84       *
85       * @see #getOrElse(Revision, String, Entry)
86       */
87      default CompletableFuture<Entry<?>> get(Revision revision, String path) {
88          return getOrElse(revision, path, null).thenApply(entry -> {
89              if (entry == null) {
90                  throw new EntryNotFoundException(revision, path);
91              }
92  
93              return entry;
94          });
95      }
96  
97      /**
98       * Performs the specified {@link Query}.
99       *
100      * @throws EntryNotFoundException if there's no entry at the path specified in the {@link Query}
101      *
102      * @see #getOrElse(Revision, Query, QueryResult)
103      */
104     default <T> CompletableFuture<QueryResult<T>> get(Revision revision, Query<T> query) {
105         return getOrElse(revision, query, null).thenApply(res -> {
106             if (res == null) {
107                 throw new EntryNotFoundException(revision, query.path());
108             }
109 
110             return res;
111         });
112     }
113 
114     /**
115      * Retrieves an {@link Entry} at the specified {@code path}. {@code other} is returned
116      *
117      * @return the {@link Entry} at the specified {@code path} if exists.
118      *         The specified {@code other} if there's no such {@link Entry}.
119      *
120      * @see #get(Revision, String)
121      */
122     default CompletableFuture<Entry<?>> getOrElse(Revision revision, String path, @Nullable Entry<?> other) {
123         validateFilePath(path, "path");
124 
125         return find(revision, path, GET_FIND_OPTIONS).thenApply(findResult -> {
126             final Entry<?> entry = findResult.get(path);
127             return entry != null ? entry : other;
128         });
129     }
130 
131     /**
132      * Performs the specified {@link Query}.
133      *
134      * @return the {@link QueryResult} on a successful query.
135      *         The specified {@code other} on a failure due to missing entry.
136      *
137      * @see #get(Revision, Query)
138      */
139     default <T> CompletableFuture<QueryResult<T>> getOrElse(
140             Revision revision, Query<T> query, @Nullable QueryResult<T> other) {
141 
142         requireNonNull(query, "query");
143         requireNonNull(revision, "revision");
144 
145         return getOrElse(revision, query.path(), null).thenApply(entry -> {
146             if (entry == null) {
147                 return other;
148             }
149 
150             final EntryType entryType = entry.type();
151 
152             if (!query.type().supportedEntryTypes().contains(entryType)) {
153                 throw new QueryException("unsupported entry type: " + entryType);
154             }
155 
156             @SuppressWarnings("unchecked")
157             final T entryContent = (T) entry.content();
158             try {
159                 return new QueryResult<>(revision, entryType, query.apply(entryContent));
160             } catch (QueryException e) {
161                 throw e;
162             } catch (Exception e) {
163                 throw new QueryException(e);
164             }
165         });
166     }
167

169      * Finds the {@link Entry}s that match the specified {@code pathPattern}.

171      * @return a {@link Map} whose value is the matching {@link Entry} and key is its path

174         return find(revision, pathPattern, ImmutableMap.of());
175     }

185     /**
186      * Query a file at two different revisions and return the diff of the two query results.
187      */
188     default CompletableFuture<Change<?>> diff(Revision from, Revision to, Query<?> query) {
189         requireNonNull(from, "from");
190         requireNonNull(to, "to");
191         requireNonNull(query, "query");
192 
193         final String path = query.path();
194         final CompletableFuture<Entry<?>> fromEntryFuture = getOrElse(from, path, null);
195         final CompletableFuture<Entry<?>> toEntryFuture = getOrElse(to, path, null);
196 
197         final CompletableFuture<Change<?>> future =
198                 CompletableFutures.combine(fromEntryFuture, toEntryFuture, (fromEntry, toEntry) -> {
199                     @SuppressWarnings("unchecked")
200                     final Query<Object> castQuery = (Query<Object>) query;
201 
202                     // Handle the case where the entry does not exist at 'from' or 'to'.
203                     if (fromEntry != null) {
204                         if (toEntry == null) {
205                             // The entry has been removed.
206                             return Change.ofRemoval(path);
207                         }
208                     } else if (toEntry != null) {
209                         // The entry has been created.
210                         final EntryType toEntryType = toEntry.type();
211                         if (!query.type().supportedEntryTypes().contains(toEntryType)) {
212                             throw new QueryException("unsupported entry type: " + toEntryType);
213                         }
214 
215                         final Object toContent = castQuery.apply(toEntry.content());
216 
217                         switch (toEntryType) {
218                             case JSON:
219                                 return Change.ofJsonUpsert(path, (JsonNode) toContent);
220                             case TEXT:
221                                 return Change.ofTextUpsert(path, (String) toContent);
222                             default:
223                                 throw new Error();
224                         }
225                     } else {
226                         // The entry did not exist both at 'from' and 'to'.
227                         throw new EntryNotFoundException(path + " (" + from + ", " + to + ')');
228                     }
229 
230                     // Handle the case where the entry exists both at 'from' and at 'to'.
231                     final EntryType entryType = fromEntry.type();
232                     if (!query.type().supportedEntryTypes().contains(entryType)) {
233                         throw new QueryException("unsupported entry type: " + entryType);
234                     }
235                     if (entryType != toEntry.type()) {
236                         throw new QueryException(
237                                 "mismatching entry type: " + entryType + " != " + toEntry.type());
238                     }
239 
240                     final Object fromContent = castQuery.apply(fromEntry.content());
241                     final Object toContent = castQuery.apply(toEntry.content());
242 
243                     switch (entryType) {
244                         case JSON:
245                             return Change.ofJsonPatch(path, (JsonNode) fromContent, (JsonNode) toContent);
246                         case TEXT:
247                             return Change.ofTextPatch(path, (String) fromContent, (String) toContent);
248                         default:
249                             throw new Error();
250                     }
251                 }).toCompletableFuture();
252 
253         return unsafeCast(future);
254     }
