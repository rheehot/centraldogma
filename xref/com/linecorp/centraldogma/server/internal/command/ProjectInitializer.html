121     static CompletableFuture<Revision> generateSampleFiles(
122             CommandExecutor executor, String projectName, String repositoryName) {
123 
124         logger.info("Generating sample files into: {}/{}", projectName, repositoryName);
125 
126         final List<Change<?>> changes = Arrays.asList(
127                 Change.ofTextUpsert("/samples/foo.txt", String.join("\n\n", SAMPLE_TEXT)),
128                 Change.ofJsonUpsert("/samples/bar.json", SAMPLE_JSON_OBJECT),
129                 Change.ofJsonUpsert("/samples/qux.json", SAMPLE_JSON_ARRAY));
130 
131         return executor.execute(Command.push(
132                 projectName, repositoryName, Revision.HEAD, Author.SYSTEM,
133                 "Add the sample files", "", Markup.PLAINTEXT, changes));
134     }
135 
136     /**
137      * Creates an internal project and repositories such as a session storage and a token storage.
138      */
139     public static void initializeInternalProject(CommandExecutor executor) {
140         try {
141             executor.execute(createProject(INTERNAL_PROJECT_NAME))
142                     .get();
143         } catch (Exception e) {
144             if (!(e.getCause() instanceof ProjectExistsException)) {
145                 throw new Error(e);
146             }
147         }
148         for (final String repo : ImmutableList.of(Project.REPO_META,
149                                                   SESSION_REPOSITORY_NAME,
150                                                   TOKEN_REPOSITORY_NAME)) {
151             try {
152                 executor.execute(createRepository(INTERNAL_PROJECT_NAME, repo))
153                         .get();
154             } catch (Exception e) {
155                 if (!(e.getCause() instanceof RepositoryExistsException)) {
156                     throw new Error(e);
157                 }
158             }
159         }
160     }
