59  /**
60   * Annotated service object for managing application tokens.
61   */
62  public class TokenService extends AbstractService {
63  
64      private static final String TOKEN_JSON_PATH = "/token.json";
65      private static final String SECRET_PREFIX = "appToken-";
66  
67      private static final TypeReference<Map<String, Token>>
68              TOKEN_MAP_TYPE_REFERENCE = new TypeReference<Map<String, Token>>() {};
69  
70      public TokenService(ProjectManager projectManager, CommandExecutor executor) {
71          super(projectManager, executor);
72      }
73  
74      /**
75       * GET /tokens
76       * Returns the list of the tokens generated before.
77       */
78      @Get("/tokens")
79      public CompletionStage<Collection<Token>> listTokens() {
80          return getTokens(Revision.HEAD)
81                  .thenApply(Map::values)
82                  .thenApply(tokens -> {
83                      final List<Token> withoutSecrets = new ArrayList<>();
84                      tokens.forEach(token -> withoutSecrets.add(token.withoutSecret()));
85                      return Collections.unmodifiableList(withoutSecrets);
86                  });
87      }
88  
89      /**
90       * POST /tokens
91       * Returns a newly-generated token belonging to the current login user.
92       */
93      @Post("/tokens")
94      public CompletionStage<Token> createToken(@Param("appId") String appId) {
95          validateFileName(appId, "appId");
96          return projectManager()
97                  .get(INTERNAL_PROJECT_NAME).repos().get(TOKEN_REPOSITORY_NAME)
98                  .normalize(Revision.HEAD)
99                  .thenCompose(revision -> {
100                     final Token token =
101                             new Token(appId, SECRET_PREFIX + UUID.randomUUID(),
102                                       AuthenticationUtil.currentUser(), new Date());
103                     return createToken0(revision, token);
104                 });
105     }
106 
107     private CompletionStage<Token> createToken0(Revision revision, Token newToken) {
108         return getTokens(revision).thenCompose(entries -> {
109             final boolean exist = entries.values().stream().anyMatch(
110                     token -> token.appId().equals(newToken.appId()));
111             if (exist) {
112                 // TODO(hyangtack) Would send 409 conflict when the following PR is merged.
113                 // https://github.com/line/armeria/pull/746
114                 throw new HttpResponseException(HttpStatus.CONFLICT);
115             }
116 
117             entries.put(newToken.secret(), newToken);
118             final User user = AuthenticationUtil.currentUser();
119             final String summary = user.name() + " generates a token: " + newToken.appId();
120             return updateTokens(revision, entries, summary).thenApply(unused -> newToken);
121         });
122     }
123 
124     /**
125      * DELETE /tokens/{id}
126      * Deletes a token of the specified ID then returns it.
127      */
128     @Delete("/tokens/{id}")
129     public CompletionStage<Token> deleteToken(@Param("id") String id) {
130         return projectManager()
131                 .get(INTERNAL_PROJECT_NAME).repos().get(TOKEN_REPOSITORY_NAME)
132                 .normalize(Revision.HEAD)
133                 .thenCompose(revision -> deleteToken0(revision, id));
134     }
135 
136     private CompletionStage<Token> deleteToken0(Revision revision, String id) {
137         return getTokens(revision).thenCompose(entries -> {
138             final Optional<Token> target = entries.values().stream()
139                                                   .filter(v -> id.equals(v.appId()))
140                                                   .findAny();
141             if (target.isPresent()) {
142                 final Token token = target.get();
143                 entries.remove(token.secret());
144                 final User user = AuthenticationUtil.currentUser();
145                 final String summary = user.name() + " deletes a token: " + token.appId();
146                 return updateTokens(revision, entries, summary).thenApply(unused -> token);
147             } else {
148                 return CompletableFuture.completedFuture(EMPTY_TOKEN);
149             }
150         });
151     }
152 
153     /**
154      * Finds a token of the specified secret.
155      */
156     public CompletionStage<Token> findToken(String secret) {
157         checkArgument(secret != null && secret.startsWith(SECRET_PREFIX),
158                       "Secret should start with " + SECRET_PREFIX + "'");
159         return projectManager()
160                 .get(INTERNAL_PROJECT_NAME).repos().get(TOKEN_REPOSITORY_NAME)
161                 .get(Revision.HEAD, Query.ofJsonPath(TOKEN_JSON_PATH, "$." + secret))
162                 .thenApply(result -> {
163                     if (result.isRemoved() ||
164                         result.type() != EntryType.JSON) {
165                         return null;
166                     }
167                     return Jackson.convertValue(result.content(), Token.class);
168                 });
169     }
170 
171     private CompletionStage<Map<String, Token>> getTokens(Revision revision) {
172         final CompletableFuture<Map<String, Token>> future = new CompletableFuture<>();
173         projectManager()
174                 .get(INTERNAL_PROJECT_NAME).repos().get(TOKEN_REPOSITORY_NAME)
175                 .get(revision, TOKEN_JSON_PATH)
176                 .thenAccept(entry -> future.complete(convert(entry)))
177                 .exceptionally(cause -> {
178                     future.complete(Maps.newHashMap());
179                     return null;
180                 });
181         return future;
182     }
183 
184     private CompletionStage<?> updateTokens(Revision revision, Map<String, Token> tokens,
185                                             String summary) {
186         return push(this, INTERNAL_PROJECT_NAME, TOKEN_REPOSITORY_NAME, revision,
187                     AuthenticationUtil.currentAuthor(), summary, "",
188                     Markup.PLAINTEXT, Change.ofJsonUpsert(TOKEN_JSON_PATH, Jackson.valueToTree(tokens)));
189     }
190 
191     private static Map<String, Token> convert(Entry<?> entry) {
192         if (entry.type() == EntryType.JSON) {
193             return Jackson.convertValue(entry.content(), TOKEN_MAP_TYPE_REFERENCE);
194         } else {
195             return Maps.newHashMap();
196         }
197     }
