95      CentralDogmaConfig(@JsonProperty(value = "dataDir", required = true) File dataDir,
96                         @JsonProperty(value = "ports", required = true)
97                         @JsonDeserialize(contentUsing = ServerPortDeserializer.class)
98                         List<ServerPort> ports,
99                         @JsonProperty("numWorkers") Integer numWorkers,
100                        @JsonProperty("maxNumConnections") Integer maxNumConnections,
101                        @JsonProperty("requestTimeoutMillis") Long requestTimeoutMillis,
102                        @JsonProperty("idleTimeoutMillis") Long idleTimeoutMillis,
103                        @JsonProperty("maxFrameLength") Integer maxFrameLength,
104                        @JsonProperty("numRepositoryWorkers") Integer numRepositoryWorkers,
105                        @JsonProperty("cacheSpec") String cacheSpec,
106                        @JsonProperty("gracefulShutdownTimeout")
107                        GracefulShutdownTimeout gracefulShutdownTimeout,
108                        @JsonProperty("webAppEnabled") Boolean webAppEnabled,
109                        @JsonProperty("mirroringEnabled") Boolean mirroringEnabled,
110                        @JsonProperty("numMirroringThreads") Integer numMirroringThreads,
111                        @JsonProperty("maxNumFilesPerMirror") Integer maxNumFilesPerMirror,
112                        @JsonProperty("maxNumBytesPerMirror") Long maxNumBytesPerMirror,
113                        @JsonProperty("replication") ReplicationConfig replicationConfig,
114                        @JsonProperty("securityEnabled") Boolean securityEnabled) {
115 
116         this.dataDir = requireNonNull(dataDir, "dataDir");
117         this.ports = ImmutableList.copyOf(requireNonNull(ports, "ports"));
118         checkArgument(!ports.isEmpty(), "ports must have at least one port.");
119         this.numWorkers = numWorkers;
120 
121         this.maxNumConnections = maxNumConnections;
122         this.requestTimeoutMillis = requestTimeoutMillis;
123         this.idleTimeoutMillis = idleTimeoutMillis;
124         this.maxFrameLength = maxFrameLength;
125         this.numRepositoryWorkers = firstNonNull(numRepositoryWorkers, DEFAULT_NUM_REPOSITORY_WORKERS);
126         checkArgument(this.numRepositoryWorkers > 0,
127                       "numRepositoryWorkers: %s (expected: > 0)", this.numRepositoryWorkers);
128         this.cacheSpec = RepositoryCache.validateCacheSpec(firstNonNull(cacheSpec, DEFAULT_CACHE_SPEC));
129         this.webAppEnabled = firstNonNull(webAppEnabled, true);
130         this.mirroringEnabled = firstNonNull(mirroringEnabled, true);
131         this.numMirroringThreads = firstNonNull(numMirroringThreads, DEFAULT_NUM_MIRRORING_THREADS);
132         checkArgument(this.numMirroringThreads > 0,
133                       "numMirroringThreads: %s (expected: > 0)", this.numMirroringThreads);
134         this.maxNumFilesPerMirror = firstNonNull(maxNumFilesPerMirror, DEFAULT_MAX_NUM_FILES_PER_MIRROR);
135         checkArgument(this.maxNumFilesPerMirror > 0,
136                       "maxNumFilesPerMirror: %s (expected: > 0)", this.maxNumFilesPerMirror);
137         this.maxNumBytesPerMirror = firstNonNull(maxNumBytesPerMirror, DEFAULT_MAX_NUM_BYTES_PER_MIRROR);
138         checkArgument(this.maxNumBytesPerMirror > 0,
139                       "maxNumBytesPerMirror: %s (expected: > 0)", this.maxNumBytesPerMirror);
140         this.gracefulShutdownTimeout = gracefulShutdownTimeout;
141         this.replicationConfig = firstNonNull(replicationConfig, ReplicationConfig.NONE);
142         this.securityEnabled = firstNonNull(securityEnabled, false);
143     }

246     static final class ServerPortSerializer extends JsonSerializer<ServerPort> {
247         @Override
248         public void serialize(ServerPort value,
249                               JsonGenerator gen, SerializerProvider serializers) throws IOException {
250 
251             final InetSocketAddress localAddr = value.localAddress();
252             final int port = localAddr.getPort();
253             final String proto = Ascii.toLowerCase(value.protocol().uriText());
254             final String host;
255 
256             if (localAddr.getAddress().isAnyLocalAddress()) {
257                 host = "*";
258             } else {
259                 final String hs = localAddr.getHostString();
260                 if (NetUtil.isValidIpV6Address(hs)) {
261                     // Try to get the platform-independent consistent IPv6 address string.
262                     host = NetUtil.toAddressString(localAddr.getAddress());
263                 } else {
264                     host = hs;
265                 }
266             }
267 
268             gen.writeStartObject();
269             gen.writeObjectFieldStart("localAddress");
270             gen.writeStringField("host", host);
271             gen.writeNumberField("port", port);
272             gen.writeEndObject();
273             gen.writeStringField("protocol", proto);
274             gen.writeEndObject();
275         }

278     static final class ServerPortDeserializer extends JsonDeserializer<ServerPort> {
279         @Override
280         public ServerPort deserialize(JsonParser p, DeserializationContext ctx) throws IOException {
281 
282             final JsonNode root = p.getCodec().readTree(p);
283             final JsonNode localAddress = root.get("localAddress");
284             if (localAddress == null || localAddress.getNodeType() != JsonNodeType.OBJECT) {
285                 return fail(ctx, root);
286             }
287 
288             final JsonNode host = localAddress.get("host");
289             if (host == null || host.getNodeType() != JsonNodeType.STRING) {
290                 return fail(ctx, root);
291             }
292 
293             final JsonNode port = localAddress.get("port");
294             if (port == null || port.getNodeType() != JsonNodeType.NUMBER) {
295                 return fail(ctx, root);
296             }
297 
298             final JsonNode proto = root.get("protocol");
299             if (proto == null || proto.getNodeType() != JsonNodeType.STRING) {
300                 return fail(ctx, root);
301             }
302 
303             final String hostVal = host.textValue();
304             final int portVal = port.intValue();
305 
306             final InetSocketAddress localAddressVal;
307             if ("*".equals(hostVal)) {
308                 localAddressVal = new InetSocketAddress(portVal);
309             } else {
310                 localAddressVal = new InetSocketAddress(hostVal, portVal);
311             }
312 
313             final SessionProtocol protoVal = SessionProtocol.of(Ascii.toUpperCase(proto.textValue()));
314 
315             return new ServerPort(localAddressVal, protoVal);
316         }
317 
318         private static ServerPort fail(DeserializationContext ctx, JsonNode root) throws JsonMappingException {
319             throw ctx.mappingException("invalid server port information: " + root);
320         }
321     }
